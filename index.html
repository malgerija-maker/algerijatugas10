01 LATIHAN 1: CONNECTION POOLING
Dartpad

Menghemat alokasi memori melalui pola **Database Connection Pooling** menggunakan Factory Constructor. Jumlah koneksi aktif dibatasi maksimal 3 koneksi, dan koneksi yang telah dilepas (*idle*) akan secara otomatis dimanfaatkan kembali. — Tema gunung mengilhami ketahanan dan efisiensi layaknya aliran sungai di pegunungan.

FILE: libconnection_factory.dart
class ConnectionFactory {
  final String id;
  final String dbName;
  final String host;
  final int port;
  bool isReleased = true;

  static const int maxConnections = 3;
  static final Map> _poolRegistry = {};

  // Private constructor
  ConnectionFactory._(this.id, this.dbName, this.host, this.port);

  // Factory constructor
  factory ConnectionFactory(String dbName, {String host = 'localhost', int port = 5432}) {
    final key = '${dbName.toLowerCase()}@$host:$port';
    
    // Initialize pool if not exists
    if (!_poolRegistry.containsKey(key)) {
      _poolRegistry[key] = [];
    }

    final connList = _poolRegistry[key]!;

    // Cari koneksi yang sedang idle (isReleased = true)
    for (var conn in connList) {
      if (conn.isReleased) {
        conn.isReleased = false;
        print(' [Pool Hit] Reusing idle connection: ${conn.id} for $key');
        return conn;
      }
    }

    // Buat koneksi baru jika belum mencapai batas maksimal
    if (connList.length < maxConnections) {
      final connIdentifier = 'conn_${dbName.toLowerCase()}_${connList.length + 1}';
      final newConnection = ConnectionFactory._(connIdentifier, dbName, host, port);
      newConnection.isReleased = false;
      connList.add(newConnection);
      print(' [Pool Miss] Created new connection: $connIdentifier untuk $key');
      return newConnection;
    }

    // Jika semua koneksi sedang aktif dan sudah mencapai batas maksimal
    throw StateError(' Batas maksimum koneksi database tercapai ($maxConnections) untuk pool $key! ');
  }

  // Method untuk menjalankan query
  void query(String sql) {
    if (isReleased) {
      throw StateError('Koneksi $id telah dilepas! Hubungkan kembali sebelum query. ');
    }
    print(' [$id] Menjalankan query: "$sql" pada database "$dbName" ($host:$port)');
  }

  // Method static untuk melepas koneksi kembali ke pool
  static void release(ConnectionFactory conn) {
    if (conn.isReleased) {
      print('⚠️ [Warning] Connection ${conn.id} already released!');
      return;
    }
    conn.isReleased = true;
    print(' [Pool Release] Connection ${conn.id} successfully returned to pool.');
  }

  // Method static untuk menampilkan status pool
  static void printStatus(String dbName, {String host = 'localhost', int port = 5432}) {
    final key = '${dbName.toLowerCase()}@$host:$port';
    final connList = _poolRegistry[key];
    
    print('\n =================================');
    print('STATUS POOL KONEKSI ($key)');
    print('=================================');
    
    if (connList == null || connList.isEmpty) {
      print('Belum ada koneksi yang terdaftar.');
    } else {
      for (var conn in connList) {
        final status = conn.isReleased ? "IDLE / TERSEDIA " : "ACTIVE / SEDANG DIGUNAKAN ";
        print(' - ${conn.id}: [$status]');
      }
      print('\nTotal koneksi: ${connList.length}/${maxConnections}');
    }
    print('=================================\n');
  }

  @override
  String toString() {
    return 'ConnectionFactory(id: $id, database: $dbName, host: $host, port: $port, status: ${isReleased ? "Idle" : "Active"})';
  }
}

void main() {
  print('=== DEMO CONNECTION POOL ===\n');
  
  try {
    // Meminta koneksi pertama
    print('1. Meminta koneksi pertama:');
    var conn1 = ConnectionFactory('testdb');
    conn1.query('SELECT * FROM users');
    ConnectionFactory.printStatus('testdb');
    
    // Meminta koneksi kedua
    print('\n2. Meminta koneksi kedua:');
    var conn2 = ConnectionFactory('testdb');
    conn2.query('SELECT * FROM products');
    ConnectionFactory.printStatus('testdb');
    
    // Meminta koneksi ketiga
    print('\n3. Meminta koneksi ketiga:');
    var conn3 = ConnectionFactory('testdb');
    conn3.query('SELECT * FROM orders');
    ConnectionFactory.printStatus('testdb');
    
    // Melepas koneksi 1
    print('\n4. Melepas koneksi 1:');
    ConnectionFactory.release(conn1);
    ConnectionFactory.printStatus('testdb');
    
    // Meminta koneksi baru (harusnya reuse conn1)
    print('\n5. Meminta koneksi baru (harusnya reuse conn1):');
    var conn4 = ConnectionFactory('testdb');
    conn4.query('SELECT * FROM logs');
    ConnectionFactory.printStatus('testdb');
    
    // Mencoba meminta koneksi ke-4 (akan error karena max 3)
    print('\n6. Mencoba meminta koneksi ke-4:');
    var conn5 = ConnectionFactory('testdb');
    conn5.query('SELECT * FROM temp');
    
  } catch (e) {
    print('\n❌ Error: $e');
  }
  
  // Contoh dengan database berbeda
  print('\n=== DATABASE BERBEDA ===\n');
  var connAnalytics = ConnectionFactory('analytics', host: '192.168.1.100', port: 3306);

  print('\n=== DEMO ERROR HANDLING ===\n');
  var tempConn = ConnectionFactory('testdb');
  tempConn.query('SELECT * FROM temp');
  ConnectionFactory.release(tempConn);
  
  try {
    tempConn.query('SELECT * FROM error'); // Ini akan error
  } catch (e) {
    print('✅ Berhasil menangkap error: $e');
  }
}

abstract class NotificationFactory {
  String get message;
  void send();

  factory NotificationFactory(String platform, String message) {
    switch (platform.toLowerCase()) {
      case 'android':
        return AndroidNotification(message);
      case 'ios':
        return IOSNotification(message);
      case 'web':
        return WebNotification(message);
      case 'windows':
        return WindowsNotification(message);
      default:
        throw ArgumentError(' Platform tidak didukung: "$platform"! Silakan pilih android, ios, web, atau windows. ');
    }
  }
}

class AndroidNotification implements NotificationFactory {
  @override
  final String message;

  AndroidNotification(this.message);

  @override
  void send() {
    print(' [Android FCM] Dispatching notification to device:');
    print('    Message payload: "$message"');
  }
}

class IOSNotification implements NotificationFactory {
  @override
  final String message;

  IOSNotification(this.message);

  @override
  void send() {
    print(' [iOS APNs] Delivering push notification:');
    print('    Message payload: "$message"');
  }
}

class WebNotification implements NotificationFactory {
  @override
  final String message;

  WebNotification(this.message);

  @override
  void send() {
    print(' [Web Push] Broadcasting service worker alert:');
    print('    Message payload: "$message"');
  }
}

class WindowsNotification implements NotificationFactory {
  @override
  final String message;

  WindowsNotification(this.message);

  @override
  void send() {
    print('🪟 [Windows WNS] Triggering desktop toast:');
    print('    Message payload: "$message"');
  }
}

import 'dart:math' as math;

abstract class ShapeFactory {
  String get type;
  double get size;
  String get color;
  void draw();

  static final Map _flyweightRegistry = {};

  factory ShapeFactory(String type, double size, String color) {
    if (size <= 0) {
      throw ArgumentError('Ukuran bentuk (size) harus lebih besar dari 0!');
    }
    if (color.trim().isEmpty) {
      throw ArgumentError('Warna bentuk tidak boleh kosong!');
    }

    final key = 'shape:${type.toLowerCase()}:${color.trim().toLowerCase()}:$size';

    if (_flyweightRegistry.containsKey(key)) {
      print('[Cache Hit] Reusing existing Shape instance: $key');
      return _flyweightRegistry[key]!;
    }

    print('[Cache Miss] Creating new Shape instance for: $key');
    ShapeFactory shapeInstance;
    switch (type.toLowerCase()) {
      case 'circle':
        shapeInstance = CircleShape._(size, color);
        break;
      case 'square':
        shapeInstance = SquareShape._(size, color);
        break;
      case 'triangle':
        shapeInstance = TriangleShape._(size, color);
        break;
      default:
        throw ArgumentError(' Tipe bentuk "$type" tidak didukung! Pilih circle, square, atau triangle. ');
    }

    _flyweightRegistry[key] = shapeInstance;
    return shapeInstance;
  }

  static int get cacheSize => _flyweightRegistry.length;

  static void clearCache() {
    _flyweightRegistry.clear();
    print('🔸 [Cache Clear] Shape cache successfully cleared.');
  }
}

class CircleShape implements ShapeFactory {

  CircleShape._(this.size, this.color);

  @override
  void draw() {
    print('🔵 [Drawing] Circle (radius: $size, color: $color)');
  }
}

class TriangleShape implements ShapeFactory {

import 'dart:math' as math;

abstract class AnimalFactory {
  String get name;
  int get age;
  String get sound;
  void speak();

  static final Map _animalRegistry = {};

  factory AnimalFactory(String type, String name, int age) {
    if (name.trim().isEmpty) {
      throw ArgumentError(' Nama hewan tidak boleh kosong! ');
    }
    if (age < 0) {
      throw ArgumentError(' Umur hewan tidak boleh kurang dari 0! ');
    }

    final key = 'animal::${type.toLowerCase()}::${name.toLowerCase().trim()}::$age';

    if (_animalRegistry.containsKey(key)) {
      print(' [Cache Hit] Reusing existing Animal instance from cache: $key');
      return _animalRegistry[key]!;
    }

    print(' [Cache Miss] Creating new Animal instance for: $key');
    AnimalFactory newAnimal;
    switch (type.toLowerCase()) {
      case 'dog':
        newAnimal = DogAnimal._(name, age);
        break;
      case 'cat':
        newAnimal = CatAnimal._(name, age);
        break;
      case 'bird':
        newAnimal = BirdAnimal._(name, age);
        break;
      default:
        throw ArgumentError('Tipe hewan "$type" tidak didukung! Pilih dog, cat, atau bird. ');
    }

    _animalRegistry[key] = newAnimal;
    return newAnimal;
  }

  static int get cacheSize => _animalRegistry.length;

  static void clearCache() {
    _animalRegistry.clear();
    print('[Cache Clear] Animal cache successfully cleared.');
  }
}

class DogAnimal implements AnimalFactory {

Refleksi

Berdasarkan implementasi keempat latihan (Connection Pooling, Notification Selector, Shape Caching, Animal Factory) dengan tema GUNUNG, berikut adalah kesimpulan utama mengenai Factory Constructor di bahasa pemrograman Dart:

📌 Ringkasan Akhir Factory Constructor:
1. Pooling & Resource Management: Factory memungkinkan pengelolaan instance terbatas dengan memeriksa objek idle sebelum membuat yang baru, mencegah resource leak.
2. Subclass Selection Dinamis: Factory dapat mengembalikan berbagai tipe turunan berdasarkan parameter runtime tanpa mengubah kode klien (polimorfisme alami).
3. Flyweight / Caching: Dengan cache statis internal, factory menghindari duplikasi objek identik, menghemat memori dan mempercepat eksekusi.
4. Validasi Terpusat: Semua logika validasi (umur negatif, nama kosong, dll) dapat dipusatkan di factory, memastikan objek yang terbentuk selalu dalam keadaan valid.
5. Fleksibilitas & Maintainability: Perubahan cara pembuatan objek cukup di dalam factory tanpa mengganggu seluruh basis kode yang menggunakan factory tersebut.

🏔️ Kesimpulan Akhir: Factory Constructor di Dart adalah pattern yang sangat esensial untuk membangun sistem yang robust, hemat memori, mudah dikembangkan, dan sesuai dengan prinsip OOP (Open/Closed, Single Responsibility). Dengan tema gunung, kita belajar bahwa seperti ekosistem pegunungan yang efisien dan teratur, factory constructor mengatur "siklus hidup" objek secara optimal, menjaga keseimbangan resource dan memori aplikasi.

01 LATIHAN 1: CONNECTION POOLING
Dartpad

Connection Pooling adalah teknik yang digunakan dalam pengembangan aplikasi (terutama yang berbasis database) untuk mengelola dan mengoptimalkan pembuatan serta penggunaan koneksi database. Teknik ini bekerja dengan menyimpan kumpulan koneksi database yang sudah dibuat sebelumnya di dalam sebuah "kolam" (pool), sehingga koneksi tersebut dapat digunakan ulang oleh berbagai permintaan dari aplikasi.

import 'dart:math';

class ConnectionFactory {
  final String id;
  final String dbName;
  final String host;
  final int port;
  bool isReleased = true;

  static const int maxConnections = 3;
  static final Map> _poolRegistry = {};

  ConnectionFactory._(this.id, this.dbName, this.host, this.port);

  factory ConnectionFactory(String dbName, {String host = 'localhost', int port = 5432}) {
    final key = '${dbName.toLowerCase()}@$host:$port';
    
    if (!_poolRegistry.containsKey(key)) {
      _poolRegistry[key] = [];
    }

    final connList = _poolRegistry[key]!;

    for (var conn in connList) {
      if (conn.isReleased) {
        conn.isReleased = false;
        print('рџЏ”пёЏ [Pool Hit] Menggunakan koneksi idle: ${conn.id} untuk $key');
        return conn;
      }
    }

    if (connList.length < maxConnections) {
      final connIdentifier = 'conn_${dbName.toLowerCase()}_${connList.length + 1}';
      final newConnection = ConnectionFactory._(connIdentifier, dbName, host, port);
      newConnection.isReleased = false;
      connList.add(newConnection);
      print('рџЊІ [Pool Miss] Membuat koneksi baru: $connIdentifier untuk $key');
      return newConnection;
    }

    throw StateError('в›°пёЏ Batas maksimum koneksi database tercapai ($maxConnections) untuk pool $key!');
  }

  void query(String sql) {
    if (isReleased) {
      throw StateError('рџЌѓ Koneksi $id telah dilepas! Hubungkan kembali sebelum query.');
    }
    print('рџЏ”пёЏ [$id] Menjalankan query: "$sql" pada database "$dbName" ($host:$port)');
  }

  static void release(ConnectionFactory conn) {
    if (conn.isReleased) {
      print('вљ пёЏ [Peringatan] Koneksi ${conn.id} sudah dilepas!');
      return;
    }
    conn.isReleased = true;
    print('рџЌѓ [Pool Release] Koneksi ${conn.id} berhasil dikembalikan ke pool.');
  }

  static void printStatus(String dbName, {String host = 'localhost', int port = 5432}) {
    final key = '${dbName.toLowerCase()}@$host:$port';
    final connList = _poolRegistry[key];
    
    print('\nв›°пёЏ =================================');
    print('STATUS POOL KONEKSI ($key)');
    print('=================================');
    
    if (connList == null || connList.isEmpty) {
      print('Belum ada koneksi yang terdaftar.');
    } else {
      for (var conn in connList) {
        final status = conn.isReleased ? "IDLE / TERSEDIA рџЌѓ" : "ACTIVE / SEDANG DIGUNAKAN рџЏ”пёЏ";
        print(' - ${conn.id}: [$status]');
      }
      print('\nTotal koneksi: ${connList.length}/$maxConnections');
    }
    print('=================================\n');
  }

  @override
  String toString() {
    return 'ConnectionFactory(id: $id, database: $dbName, host: $host, port: $port, status: ${isReleased ? "Idle" : "Active"})';
  }
}

03 LATIHAN 3: SHAPE CACHING (FLYWEIGHT)

Flyweight Pattern adalah pola desain struktural yang bertujuan untuk meminimalkan penggunaan memori dengan cara berbagi (sharing) data sebanyak mungkin antar objek yang serupa. Pola ini sangat berguna ketika Anda perlu membuat banyak objek dalam jumlah sangat besar, tetapi hanya sedikit variasi data di antara mereka.

abstract class ShapeFactory {
  String get type;
  double get size;
  String get color;
  void draw();

  static final Map _flyweightRegistry = {};

  factory ShapeFactory(String type, double size, String color) {
    if (size <= 0) throw ArgumentError('Ukuran harus > 0!');
    if (color.trim().isEmpty) throw ArgumentError('Warna tidak boleh kosong');
    final key = 'shape:${type.toLowerCase()}:${color.trim().toLowerCase()}:$size';
    if (_flyweightRegistry.containsKey(key)) {
      print('рџЏ”пёЏ [Cache Hit] Menggunakan ulang bentuk: $key');
      return _flyweightRegistry[key]!;
    }
    print('рџЊІ [Cache Miss] Membuat bentuk baru: $key');
    ShapeFactory shape;
    switch (type.toLowerCase()) {
      case 'circle': shape = CircleShape._(size, color); break;
      case 'square': shape = SquareShape._(size, color); break;
      case 'triangle': shape = TriangleShape._(size, color); break;
      default: throw ArgumentError('Tipe tidak didukung');
    }
    _flyweightRegistry[key] = shape;
    return shape;
  }
  static int get cacheSize => _flyweightRegistry.length;
  static void clearCache() => _flyweightRegistry.clear();
}

class CircleShape implements ShapeFactory {
  @override final String type = 'Circle';
  @override final double size;
  @override final String color;
  CircleShape._(this.size, this.color);
  @override void draw() => print('рџ”µ Lingkaran (radius: $size, warna: $color)');
}

  SquareShape._(this.size, this.color);
  @override void draw() => print('рџџ© Persegi (sisi: $size, warna: $color)');

    if (name.trim().isEmpty) throw ArgumentError('вќЊ Nama hewan tidak boleh kosong');
    if (age < 0) throw ArgumentError('вќЊ Umur hewan tidak boleh negatif');
    
    final key = 'animal::${type.toLowerCase()}::${name.toLowerCase().trim()}::$age';
    
    if (_animalRegistry.containsKey(key)) {
      print('рџЏ”пёЏ [Cache Hit] Memakai hewan dari cache: $key');
      return _animalRegistry[key]!;
    }
    
    print('рџЊї [Cache Miss] Membuat hewan baru: $key');
    AnimalFactory animal;
    switch (type.toLowerCase()) {
      case 'dog': animal = DogAnimal._(name, age); break;
      case 'cat': animal = CatAnimal._(name, age); break;
      case 'bird': animal = BirdAnimal._(name, age); break;
      default: throw ArgumentError('вќЊ Tipe hewan "$type" tidak dikenal! Pilih: dog, cat, bird');
    }
    _animalRegistry[key] = animal;
    return animal;
  }
  
  static int get cacheSize => _animalRegistry.length;
  static void clearCache() => _animalRegistry.clear();
}

01 LATIHAN 1: CONNECTION POOLING
Dartpad

Connection Pooling adalah teknik yang digunakan dalam pengembangan aplikasi (terutama yang berbasis database) untuk mengelola dan mengoptimalkan pembuatan serta penggunaan koneksi database. Teknik ini bekerja dengan menyimpan kumpulan koneksi database yang sudah dibuat sebelumnya di dalam sebuah "kolam" (pool), sehingga koneksi tersebut dapat digunakan ulang oleh berbagai permintaan dari aplikasi

FILE: libconnection_factory.dart
class ConnectionFactory {
  final String id;
  final String dbName;
  final String host;
  final int port;
  bool isReleased = true;

  static const int maxConnections = 3;
  static final Map> _poolRegistry = {};

  // Private constructor
  ConnectionFactory._(this.id, this.dbName, this.host, this.port);

  // Factory constructor
  factory ConnectionFactory(String dbName, {String host = 'localhost', int port = 5432}) {
    final key = '${dbName.toLowerCase()}@$host:$port';
    
    if (!_poolRegistry.containsKey(key)) {
      _poolRegistry[key] = [];
    }

    final connList = _poolRegistry[key]!;

    for (var conn in connList) {
      if (conn.isReleased) {
        conn.isReleased = false;
        print('🏔️ [Pool Hit] Reusing idle connection: ${conn.id} for $key');
        return conn;
      }
    }

    if (connList.length < maxConnections) {
      final connIdentifier = 'conn_${dbName.toLowerCase()}_${connList.length + 1}';
      final newConnection = ConnectionFactory._(connIdentifier, dbName, host, port);
      newConnection.isReleased = false;
      connList.add(newConnection);
      print('🌲 [Pool Miss] Created new connection: $connIdentifier untuk $key');
      return newConnection;
    }

    throw StateError('⛰️ Batas maksimum koneksi database tercapai ($maxConnections) untuk pool $key!');
  }

  void query(String sql) {
    if (isReleased) {
      throw StateError('🌿 Koneksi $id telah dilepas! Hubungkan kembali sebelum query.');
    }
    print('🏔️ [$id] Menjalankan query: "$sql" pada database "$dbName" ($host:$port)');
  }

  static void release(ConnectionFactory conn) {
    if (conn.isReleased) {
      print('⚠️ [Warning] Connection ${conn.id} already released!');
      return;
    }
    conn.isReleased = true;
    print('🍃 [Pool Release] Connection ${conn.id} successfully returned to pool.');
  }

  static void printStatus(String dbName, {String host = 'localhost', int port = 5432}) {
    final key = '${dbName.toLowerCase()}@$host:$port';
    final connList = _poolRegistry[key];
    
    print('\n⛰️ =================================');
    print('STATUS POOL KONEKSI ($key)');
    print('=================================');
    
    if (connList == null || connList.isEmpty) {
      print('Belum ada koneksi yang terdaftar.');
    } else {
      for (var conn in connList) {
        final status = conn.isReleased ? "IDLE / TERSEDIA 🍃" : "ACTIVE / SEDANG DIGUNAKAN 🏔️";
        print(' - ${conn.id}: [$status]');
      }
      print('\nTotal koneksi: ${connList.length}/${maxConnections}');
    }
    print('=================================\n');
  }

  @override
  String toString() {
    return 'ConnectionFactory(id: $id, database: $dbName, host: $host, port: $port, status: ${isReleased ? "Idle" : "Active"})';
  }
}

void main() {
  print('=== DEMO CONNECTION POOL ===\n');
  try {
    var conn1 = ConnectionFactory('testdb');
    conn1.query('SELECT * FROM users');
    ConnectionFactory.printStatus('testdb');
    
    var conn2 = ConnectionFactory('testdb');
    conn2.query('SELECT * FROM products');
    ConnectionFactory.printStatus('testdb');
    
    var conn3 = ConnectionFactory('testdb');
    conn3.query('SELECT * FROM orders');
    ConnectionFactory.printStatus('testdb');
    
    ConnectionFactory.release(conn1);
    ConnectionFactory.printStatus('testdb');
    
    var conn4 = ConnectionFactory('testdb');
    conn4.query('SELECT * FROM logs');
    ConnectionFactory.printStatus('testdb');
    
    var conn5 = ConnectionFactory('testdb');
    conn5.query('SELECT * FROM temp');
  } catch (e) {
    print('\n❌ Error: $e');
  }
}

02 LATIHAN 2: PLATFORM NOTIFICATION SELECTOR
Dartpad

Subclass Selection adalah proses menentukan atau memilih subclass mana yang akan diinstansiasi (dibuat objeknya) pada saat runtime, berdasarkan kondisi atau parameter tertentu. Pola ini sering muncul ketika superclass memiliki beberapa subclass turunan, dan aplikasi harus secara dinamis memilih salah satu yang sesuai dengan kebutuhan.

FILE: libnotification_factory.dart
abstract class NotificationFactory {
  String get message;
  void send();

  factory NotificationFactory(String platform, String message) {
    switch (platform.toLowerCase()) {
      case 'android':
        return AndroidNotification(message);
      case 'ios':
        return IOSNotification(message);
      case 'web':
        return WebNotification(message);
      case 'windows':
        return WindowsNotification(message);
      default:
        throw ArgumentError('⛰️ Platform tidak didukung: "$platform"! Pilih android, ios, web, atau windows.');
    }
  }
}

class AndroidNotification implements NotificationFactory {
  @override
  final String message;
  AndroidNotification(this.message);
  @override
  void send() {
    print('📱 [Android FCM] Mengirim notifikasi gunung:');
    print('   💌 Pesan: "$message"');
  }
}

03 LATIHAN 3: SHAPE CACHING (FLYEWIGHT)
Dartpad

Flyweight Pattern adalah pola desain struktural yang bertujuan untuk meminimalkan penggunaan memori dengan cara berbagi (sharing) data sebanyak mungkin antar objek yang serupa. Pola ini sangat berguna ketika Anda perlu membuat banyak objek dalam jumlah sangat besar,tetapi hanya sedikit variasi data di antara mereka.

FILE: libshape_factory.dart
abstract class ShapeFactory {
  String get type;
  double get size;
  String get color;
  void draw();

  static final Map _flyweightRegistry = {};

  factory ShapeFactory(String type, double size, String color) {
    if (size <= 0) throw ArgumentError('Ukuran harus > 0!');
    if (color.trim().isEmpty) throw ArgumentError('Warna tidak boleh kosong');
    final key = 'shape:${type.toLowerCase()}:${color.trim().toLowerCase()}:$size';
    if (_flyweightRegistry.containsKey(key)) {
      print('🏔️ [Cache Hit] Menggunakan ulang bentuk: $key');
      return _flyweightRegistry[key]!;
    }
    print('🌲 [Cache Miss] Membuat bentuk baru: $key');
    ShapeFactory shape;
    switch (type.toLowerCase()) {
      case 'circle': shape = CircleShape._(size, color); break;
      case 'square': shape = SquareShape._(size, color); break;
      case 'triangle': shape = TriangleShape._(size, color); break;
      default: throw ArgumentError('Tipe tidak didukung');
    }
    _flyweightRegistry[key] = shape;
    return shape;
  }
  static int get cacheSize => _flyweightRegistry.length;
}

class CircleShape implements ShapeFactory {
  @override final String type = 'Circle';
  @override final double size;
  @override final String color;
  CircleShape._(this.size, this.color);
  @override void draw() => print('🔵 Lingkaran (radius: $size, warna: $color)');
}
class SquareShape implements ShapeFactory {
  @override final String type = 'Square';
  @override final double size;
  @override final String color;
  SquareShape._(this.size, this.color);
  @override void draw() => print('🟩 Persegi (sisi: $size, warna: $color)');
}
class TriangleShape implements ShapeFactory {
  @override final String type = 'Triangle';
  @override final double size;
  @override final String color;
  TriangleShape._(this.size, this.color);
  @override void draw() => print('🔺 Segitiga (dimensi: $size, warna: $color)');
}

04 CHALLENGE: ANIMAL FACTORY
Challenge

Factory Constructor adalah metode (biasanya static method atau method khusus) di dalam kelas factory yang bertanggung jawab untuk membuat dan mengembalikan instance dari berbagai subclass berdasarkan parameter input. Di sini, AnimalFactory bertindak sebagai "pabrik" yang memproduksi berbagai jenis hewan.

FILE: libanimal_factory.dart
abstract class AnimalFactory {
  String get name;
  int get age;
  String get sound;
  void speak();
  static final Map _animalRegistry = {};

  factory AnimalFactory(String type, String name, int age) {
    if (name.trim().isEmpty) throw ArgumentError('Nama hewan tidak boleh kosong');
    if (age < 0) throw ArgumentError('Umur hewan tidak boleh negatif');
    final key = 'animal::${type.toLowerCase()}::${name.toLowerCase().trim()}::$age';
    if (_animalRegistry.containsKey(key)) {
      print('🏔️ [Cache Hit] Memakai hewan dari cache: $key');
      return _animalRegistry[key]!;
    }
    print('🌿 [Cache Miss] Hewan baru didaftarkan: $key');
    AnimalFactory animal;
    switch (type.toLowerCase()) {
      case 'dog': animal = DogAnimal._(name, age); break;
      case 'cat': animal = CatAnimal._(name, age); break;
      case 'bird': animal = BirdAnimal._(name, age); break;
      default: throw ArgumentError('Tipe hewan tidak dikenal');
    }
    _animalRegistry[key] = animal;
    return animal;
  }
  static int get cacheSize => _animalRegistry.length;
}

class DogAnimal implements AnimalFactory {
  @override final String name;
  @override final int age;
  @override final String sound = 'Guk Guk!';
  DogAnimal._(this.name, this.age);
  @override void speak() => print('🐶 Anjing $name ($age tahun): $sound');
}
class CatAnimal implements AnimalFactory {
  @override final String name;
  @override final int age;
  @override final String sound = 'Meong~';
  CatAnimal._(this.name, this.age);
  @override void speak() => print('🐱 Kucing $name ($age tahun): $sound');
}
class BirdAnimal implements AnimalFactory {
  @override final String name;
  @override final int age;
  @override final String sound = 'Cit cit';
  BirdAnimal._(this.name, this.age);
  @override void speak() => print('🐦 Burung $name ($age tahun): $sound');
}
